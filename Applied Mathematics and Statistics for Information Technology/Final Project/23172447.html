def mat_transpose(matrix):
    return [list(row) for row in zip(*matrix)]

def mat_mult(A, B):
    result = [[0 for _ in range(len(B[0]))] for _ in range(len(A))]
    for i in range(len(A)):
        for j in range(len(B[0])):
            s = 0
            for k in range(len(B)):
                s += A[i][k] * B[k][j]
            result[i][j] = s 
    return result

def mat_inverse(matrix):
    n = len(matrix)
    mat = [row[:] for row in matrix] 
    identity = [[float(i == j) for j in range(n)] for i in range(n)]

    for i in range(n):
        diag = mat[i][i]
        for j in range(n):
            mat[i][j] /= diag
            identity[i][j] /= diag
        for k in range(n):
            if i != k:
                factor = mat[k][i]
                for j in range(n):
                    mat[k][j] -= factor * mat[i][j]
                    identity[k][j] -= factor * identity[i][j]
    return identity 

def train_ols(X, y):
    X_T = mat_transpose(X)
    XTX = mat_mult(X_T, X)
    XTy = mat_mult(X_T, [[v] for v in y])
    XTX_inv = mat_inverse(XTX)
    beta = mat_mult(XTX_inv, XTy)
    return beta

def predict(X, beta):
    return [sum(beta[j][0] * row[j] for j in range(len(beta))) for row in X]

Trong :

  • $y$ l vector gi tr thc t.
  • $\hat{y}$ l vector gi tr d on, c tnh bng hm predict(X, beta).

Chun L2 ny phn nh mc sai lch tng th gia d on v thc t:

  • Chun cng nh m hnh d on cng chnh xc.
  • Chun cng ln sai s tng th cng cao.
In[164]:
def residual_norm_loss(y_true, X, beta):
    y_pred = predict(X, beta)
    residuals = [(yt - yp) for yt, yp in zip(y_true, y_pred)]
    norm = (sum(r ** 2 for r in residuals)) ** 0.5
    return norm

Cu d

Xy dng m hnh hi quy tuyn tnh a bin vi bin gc v tt c cc bin tng tc

  • Bin ph thuc: sprice (gi bn nh).
  • Bin c lp:
    • Cc bin gc: livarea, beds, baths, age, lgelot, pool.
    • Tt c cc bin tng tc c th to ra t bin gc (t hp 2 bin tr ln).

M hnh hi quy tuyn tnh a bin c biu din di dng:

$$ y = \beta_0 + \sum_{i=1}^p \beta_i x_i + \sum_{j=1}^q \beta_j z_j $$

trong :

  • $y$ l bin ph thuc (sprice),
  • $x_i$ l cc bin gc,
  • $z_j$ l cc bin tng tc c to ra t tch cc bin gc,
  • $\beta_0$ l h s chn,
  • $\beta_i$, $\beta_j$ l h s hi quy tng ng.

Cc bc thc hin

Bc 1: Chun b d liu

  • To bin tng tc cho mi t hp bin gc t 2 bin tr ln bng cch nhn tng cp, b 3,... cc bin li vi nhau.
  • Tp hp c trng cui cng gm: cc bin gc + cc bin tng tc.

Bc 2: To ma trn c trng v vector mc tiu

  • Ma trn $X$ gm:

    • Mt ct intercept (hng s 1).
    • Cc bin gc: livarea, beds, baths, age, lgelot, pool.
    • Tt c cc bin tng tc gia cc bin gc (t hp t 2 bin tr ln).

  • Vector mc tiu $y$ l gi tr sprice (gi bn nh).

Bc 3: Hun luyn m hnh

  • S dng phng php bnh phng nh nht OLS tm h s $\hat{\beta}$:
    $$ \hat{\beta} = (X^T X)^{-1} X^T y $$

  • In h s hi quy v phng trnh hi quy hiu v din gii m hnh. Bc 4: nh gi m hnh

  • Tnh gi tr d on t m hnh:

    $$ \hat{y} = X \hat{\beta} $$

  • Tnh chun L2 ca vector phn d ( ln vector sai s) nh gi chnh xc ca m hnh:

    $$ \| y - \hat{y} \| = \sqrt{ \sum_{i=1}^n (y_i - \hat{y}_i)^2 } $$

  • Chun L2 cng nh m hnh d on cng chnh xc.

Bc 5: Trc quan ha

  • V biu phn tn gia gi tr thc t sprice v gi tr d on $\hat{y}$.
  • Thm ng cho $y=x$ nh gi trc quan chnh xc m hnh.
  • Nu cc im gn ng cho, m hnh d on tt.

def build_count_matrix(data, states=['K','B','T']):
    n = len(states)
    count_matrix = [[0]*n for _ in range(n)]
    state_to_index = {s:i for i,s in enumerate(states)}

    for idx in range(len(data)-1):
        start = state_to_index[data[idx]]     # trng thi xut pht (j)
        end = state_to_index[data[idx+1]]     # trng thi n (i)
        count_matrix[end][start] += 1         # tng count cho hng i, ct j
    
    return count_matrix

def build_transition_matrix_col_norm(count_matrix):
    n = len(count_matrix)
    col_sums = [0]*n
    for j in range(n):
        s = 0
        for i in range(n):
            s += count_matrix[i][j]
        col_sums[j] = s
    
    P = []
    for i in range(n):
        row = []
        for j in range(n):
            if col_sums[j] == 0:
                row.append(1/n)  # gi s u nu ct rng
            else:
                row.append(count_matrix[i][j]/col_sums[j])
        P.append(row)
    return P

def multiply_matrix_vector(P, v):
    n = len(P)
    result = [0]*n
    for i in range(n):
        s = 0
        for j in range(n):
            s += P[i][j]*v[j]
        result[i] = s
    return result

# Hm tnh phn phi ngy k: pi(k) = P^k * pi(0)
def prop_distribution(P, k, initial_distribution=None):
    if initial_distribution is None:
        initial_distribution = [1/3]*len(P)  # vector ct u ban u
    
    dist = initial_distribution[:]
    for _ in range(k):
        dist = multiply_matrix_vector(P, dist)
    return dist

Cu e

Tm phn phi dng ca xch Markov

Phn phi dng $\pi$ ca ma trn chuyn trng thi $P$ l vector xc sut tha mn:

$$ P \pi = \pi $$

iu ny tng ng vi h phng trnh:

$$ (P - I) \pi = 0 $$

trong $I$ l ma trn n v.

V $\pi$ l vector xc sut, cc phn t $\pi_i$ phi tha mn:

$$ \sum_{i} \pi_i = 1 $$

Do , gii h, ta thay th mt trong cc phng trnh ca $(P - I) \pi = 0$ bng phng trnh tng xc sut bng 1 h c nghim duy nht.

Thut ton gii h phng trnh

y, ta s dng phng php kh Gauss gii h phng trnh tuyn tnh:

  • To ma trn m rng $[A|b]$ vi $A = P - I$ v vector $b$ c tt c gi tr 0 ngoi tr phn t cui cng l 1 (tng ng vi rng buc tng xc sut).
  • Thc hin kh Gauss a ma trn v dng tam gic trn.
  • Tnh nghim $\pi$ t ma trn bin i.

Kt qu

Sau khi gii, ta thu c phn phi dng $\pi$, i din cho xc sut lu di m h thng giao thng s mi trng thi.

def gauss_solve(A, b):
    """
    Gii h Ax = b bng phng php kh Gauss.
    A l ma trn vung, b l vector ct.
    """
    n = len(A)
    # To ma trn m rng [A|b]
    M = [A[i][:] + [b[i]] for i in range(n)]

    for i in range(n):
        # Tm pivot
        max_row = max(range(i, n), key=lambda r: abs(M[r][i]))
        if abs(M[max_row][i]) < 1e-15:
            raise ValueError("Ma trn suy bin, khng th gii")

        # i dng
        M[i], M[max_row] = M[max_row], M[i]

        # Chun ha pivot
        pivot = M[i][i]
        for j in range(i, n+1):
            M[i][j] /= pivot

        # Kh cc dng khc
        for r in range(n):
            if r != i:
                factor = M[r][i]
                for c in range(i, n+1):
                    M[r][c] -= factor * M[i][c]

    # Tr v nghim
    x = [M[i][n] for i in range(n)]
    return x

def solve_steady_state(P):
    n = len(P)
    # To ma trn A = P - I
    A = [[P[i][j] for j in range(n)] for i in range(n)]
    for i in range(n):
        A[i][i] -= 1

    # Thay dng cui bng rng buc sum pi_i = 1
    A[-1] = [1]*n
    b = [0]*(n-1) + [1]

    pi = gauss_solve(A, b)
    return pi
